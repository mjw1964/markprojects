    function formatCountdown(ms) {

      if (ms <= 0) return "Now";


      const totalMinutes = Math.floor(ms / 60000);

      const days = Math.floor(totalMinutes / 1440);

      const hours = Math.floor((totalMinutes % 1440) / 60);

      const minutes = totalMinutes % 60;


      if (days > 0) {

        return `${days}d ${hours}h`;

      }


      if (hours > 0) {

        return `${hours}h ${minutes}m`;

      }


      return `${minutes}m`;

    }


    function formatMeetingTime(date) {

      return date.toLocaleString("en-US", {

        weekday: "short",

        hour: "numeric",

        minute: "2-digit",

        hour12: true

      });

    }


    function updateNextMeeting() {

      const now = new Date();


      const upcoming = meetings

        .map(meeting => ({

          ...meeting,

          date: new Date(meeting.start)

        }))

        .filter(meeting => meeting.date > now)

        .sort((a, b) => a.date - b.date);


      const countdownEl = document.getElementById("nextCountdown");

      const titleEl = document.getElementById("nextTitle");

      const detailEl = document.getElementById("nextDetail");


      if (!upcoming.length) {

        countdownEl.textContent = "Clear";

        titleEl.textContent = "No upcoming meetings";

        detailEl.textContent = "Calendar signal quiet";

        return;

      }


      const next = upcoming[0];

      const diff = next.date - now;


      countdownEl.textContent = formatCountdown(diff);

      titleEl.textContent = next.title;

      detailEl.textContent = formatMeetingTime(next.date);

    }


    updateNextMeeting();

    setInterval(updateNextMeeting, 60000);
